    <script>
        const fullVerbPool = [

            { verb: 'finish', type: 'gerund', hint: 'Talk about completing a task.' },
            { verb: 'quit', type: 'gerund', hint: 'Giving up a bad habit.' },
            { verb: 'suggest', type: 'gerund', hint: 'Recommending an action to someone.' },
            { verb: 'miss', type: 'gerund', hint: 'Something you miss from your home country.' },
            { verb: 'keep', type: 'gerund', hint: 'Continuing to do something.' },
            { verb: 'mind', type: 'gerund', hint: 'Used in questions/negatives (Do you mind...?)' },
            { verb: 'postpone', type: 'gerund', hint: 'Delaying an event.' },
            { verb: 'risk', type: 'gerund', hint: 'Doing something dangerous.' },
            { verb: 'admit', type: 'gerund', hint: 'Saying you did something wrong.' },
            { verb: 'deny', type: 'gerund', hint: 'Saying you did NOT do something.' },
            { verb: 'imagine', type: 'gerund', hint: 'Picture a future scenario.' },
            { verb: 'appreciate', type: 'gerund', hint: 'Being thankful for an action.' },
            { verb: 'recommend', type: 'gerund', hint: 'Give advice on a good action.' },
            { verb: 'resist', type: 'gerund', hint: 'Try not to do something tempting.' },

            // INFINITIVE ONLY
            { verb: 'hope', type: 'infinitive', hint: 'A wish for the future.' },
            { verb: 'decide', type: 'infinitive', hint: 'A choice you made recently.' },
            { verb: 'plan', type: 'infinitive', hint: 'Arrangements for your next vacation.' },
            { verb: 'promise', type: 'infinitive', hint: 'A commitment to a friend.' },
            { verb: 'agree', type: 'infinitive', hint: 'Consenting to a request.' },
            { verb: 'offer', type: 'infinitive', hint: 'Volunteering to help someone.' },
            { verb: 'refuse', type: 'infinitive', hint: 'Saying no to an invitation.' },
            { verb: 'wait', type: 'infinitive', hint: 'Anticipating an event.' },
            { verb: 'want', type: 'infinitive', hint: 'A basic desire.' },
            { verb: 'need', type: 'infinitive', hint: 'A requirement for school.' },
            { verb: 'learn', type: 'infinitive', hint: 'Acquiring a new skill.' },
            { verb: 'fail', type: 'infinitive', hint: 'Not succeeding in a task.' },
            { verb: 'seem', type: 'infinitive', hint: 'Giving an impression.' },
            { verb: 'appear', type: 'infinitive', hint: 'Describing how someone looks.' },
            { verb: 'deserve', type: 'infinitive', hint: 'Being worthy of something.' },
            { verb: 'pretend', type: 'infinitive', hint: 'Acting like something is true.' },
            { verb: 'tend', type: 'infinitive', hint: 'Something that happens usually.' },
            { verb: 'afford', type: 'infinitive', hint: 'Having enough money for an action.' },

            // BOTH (Same meaning)
            { verb: 'like', type: 'both', hint: 'A general preference.' },
            { verb: 'love', type: 'both', hint: 'A strong preference.' },
            { verb: 'hate', type: 'both', hint: 'A strong dislike.' },
            { verb: 'start', type: 'both', hint: 'Beginning an action.' },
            { verb: 'continue', type: 'both', hint: 'Keeping on with a task.' },

            // TRICKY (Change meaning)
            { verb: 'stop', type: 'tricky', hint: 'Quit a habit vs. Pause to do something.' },
            { verb: 'remember', type: 'tricky', hint: 'A memory of the past vs. A task for the future.' },
            { verb: 'forget', type: 'tricky', hint: 'Losing a memory vs. Failing to do a task.' },
            { verb: 'try', type: 'tricky', hint: 'Attempting something difficult vs. Experimenting.' },
            { verb: 'regret', type: 'tricky', hint: 'Feeling sorry for the past vs. Telling bad news.' }
        ];

        // SESSION SETTINGS
        const SESSION_LENGTH = 10; 
        let currentSessionPool = [];
        let currentIndex = 0;
        let score = 0;

        const ui = {
            targetVerb: document.getElementById('target-verb'),
            promptHint: document.getElementById('prompt-hint'),
            userInput: document.getElementById('user-input'),
            feedbackBox: document.getElementById('feedback-box'),
            feedbackText: document.getElementById('feedback-text'),
            explanationText: document.getElementById('explanation-text'),
            feedbackIcon: document.getElementById('feedback-icon'),
            checkBtn: document.getElementById('check-btn'),
            nextBtn: document.getElementById('next-btn'),
            retryBtn: document.getElementById('retry-btn'),
            resetBtn: document.getElementById('reset-current-btn'),
            badge: document.getElementById('category-badge'),
            scoreLabel: document.getElementById('score-correct'),
            countLabel: document.getElementById('current-count'),
            progressBar: document.getElementById('progress-bar')
        };

        // Fisher-Yates Shuffle
        function shuffle(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        function startNewSession() {
            const shuffled = shuffle([...fullVerbPool]);
            currentSessionPool = shuffled.slice(0, SESSION_LENGTH);
            currentIndex = 0;
            score = 0;
            ui.scoreLabel.innerText = "0";
            initQuestion();
        }

        function initQuestion() {
            const data = currentSessionPool[currentIndex];
            ui.targetVerb.innerText = data.verb;
            ui.promptHint.innerText = data.hint;
            ui.userInput.value = '';
            ui.userInput.disabled = false;
            ui.userInput.className = "w-full p-4 border-2 border-slate-200 rounded-xl focus:ring-4 focus:ring-blue-100 focus:border-blue-500 transition-all outline-none h-32 text-lg";
            ui.feedbackBox.classList.add('hidden');
            ui.checkBtn.classList.remove('hidden');
            ui.nextBtn.classList.add('hidden');
            ui.retryBtn.classList.add('hidden');
            ui.resetBtn.classList.add('hidden');
            
            const colors = {
                gerund: 'bg-green-100 text-green-700',
                infinitive: 'bg-blue-100 text-blue-700',
                tricky: 'bg-red-100 text-red-700',
                both: 'bg-orange-100 text-orange-700'
            };
            ui.badge.className = `px-3 py-1 rounded-full text-xs font-bold uppercase tracking-wider mb-2 inline-block ${colors[data.type]}`;
            ui.badge.innerText = data.type === 'tricky' ? '‚ö†Ô∏è Check Meaning' : data.type;
            
            ui.countLabel.innerText = `${currentIndex + 1} / ${SESSION_LENGTH}`;
            ui.progressBar.style.width = `${(currentIndex / SESSION_LENGTH) * 100}%`;
            ui.userInput.focus();
        }

        function checkAnswer() {
            const text = ui.userInput.value.trim().toLowerCase();
            const data = currentSessionPool[currentIndex];
            const verb = data.verb.toLowerCase();

            if (!text.includes(verb)) {
                showFeedback('error', 'Missing Verb', `You must use the word "${data.verb}" (or its variations like ${verb}s, ${verb}ed, ${verb}ing).`);
                return;
            }

            const regex = new RegExp(`${verb}(?:s|ed|ing)?\\s+([a-z'\\-]+)`, 'i');
            const match = text.match(regex);

            const isGerund = match ? match[1].endsWith('ing') : false;
            const isInfinitive = text.includes(`${verb} to `) || text.includes(`${verb}s to `) || text.includes(`${verb}ed to `) || text.includes(`${verb}ing to `);

            let isCorrect = false;
            let msg = "";

            if (data.type === 'gerund') {
                isCorrect = isGerund;
                msg = isCorrect ? "Excellent! Always follow this verb with a gerund." : `Oops! "${data.verb}" must be followed by -ing (e.g., ${data.verb} studying).`;
            } else if (data.type === 'infinitive') {
                isCorrect = isInfinitive;
                msg = isCorrect ? "Great! You correctly used 'to' + verb." : `Wait! "${data.verb}" needs an infinitive (e.g., ${data.verb} to study).`;
            } else if (data.type === 'tricky' || data.type === 'both') {
                isCorrect = isGerund || isInfinitive;
                msg = data.type === 'tricky' ? "Grammar is fine, but check the meaning!" : "Good! Both forms are acceptable here.";
            }

            if (isCorrect) {
                score++;
                ui.scoreLabel.innerText = score;
                showFeedback('success', "Correct!", msg);
                ui.userInput.disabled = true;
                ui.checkBtn.classList.add('hidden');
                ui.retryBtn.classList.add('hidden');
                ui.resetBtn.classList.add('hidden');
                ui.nextBtn.classList.remove('hidden');
            } else {
                showFeedback('error', "Incorrect Pattern", msg);
                ui.checkBtn.classList.add('hidden');
                ui.retryBtn.classList.remove('hidden');
                ui.resetBtn.classList.remove('hidden');
            }
        }

        function showFeedback(type, title, msg) {
            ui.feedbackBox.classList.remove('hidden');
            ui.feedbackText.innerText = title;
            ui.explanationText.innerText = msg;
            if (type === 'success') {
                ui.feedbackBox.className = "mt-4 p-4 rounded-xl border-2 correct text-green-800 animate-pop";
                ui.userInput.classList.add('correct');
                ui.feedbackIcon.innerHTML = '‚úÖ';
            } else {
                ui.feedbackBox.className = "mt-4 p-4 rounded-xl border-2 incorrect text-red-800 animate-pop";
                ui.userInput.classList.add('incorrect');
                ui.feedbackIcon.innerHTML = '‚ùå';
            }
        }

        function handleRetry() {
            ui.feedbackBox.classList.add('hidden');
            ui.retryBtn.classList.add('hidden');
            ui.resetBtn.classList.add('hidden');
            ui.checkBtn.classList.remove('hidden');
            ui.userInput.classList.remove('incorrect');
            ui.userInput.focus();
        }

        function resetCurrent() {
            ui.userInput.value = '';
            handleRetry();
        }

        function nextQuestion() {
            currentIndex++;
            if (currentIndex < SESSION_LENGTH) {
                initQuestion();
            } else {
                finishGame();
            }
        }

        function finishGame() {
            ui.progressBar.style.width = "100%";
            const percent = Math.round((score / SESSION_LENGTH) * 100);
            let message = percent >= 80 ? "Fantastic Work!" : percent >= 50 ? "Good Job!" : "Keep Practicing!";
            
            document.getElementById('game-container').innerHTML = `
                <div class="text-center py-10 animate-pop">
                    <div class="text-6xl mb-4">${percent >= 80 ? 'üèÜ' : 'üìö'}</div>
                    <h2 class="text-3xl font-bold text-slate-800 mb-2">${message}</h2>
                    <p class="text-slate-500 mb-8 text-lg">Score: ${score} / ${SESSION_LENGTH} (${percent}%)</p>
                    <button onclick="location.reload()" class="bg-blue-600 text-white px-10 py-4 rounded-xl font-bold shadow-lg hover:bg-blue-700 transition-all">Start New Session</button>
                </div>
            `;
        }

        function toggleReference() {
            document.getElementById('ref-table').classList.toggle('hidden');
        }

        ui.checkBtn.addEventListener('click', checkAnswer);
        ui.nextBtn.addEventListener('click', nextQuestion);
        ui.retryBtn.addEventListener('click', handleRetry);
        ui.resetBtn.addEventListener('click', resetCurrent);

        ui.userInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                if (!ui.checkBtn.classList.contains('hidden')) checkAnswer();
                else if (!ui.nextBtn.classList.contains('hidden')) nextQuestion();
                else if (!ui.retryBtn.classList.contains('hidden')) handleRetry();
            }
        });

        startNewSession();
    </script>
